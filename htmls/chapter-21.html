import numpy as np

# For F=<-y/2, x/2>, circulation around unit circle should be pi.
t = np.linspace(0, 2*np.pi, 10001)
x, y = np.cos(t), np.sin(t)
dxdt, dydt = -np.sin(t), np.cos(t)
P, Q = -y/2, x/2
print(np.trapz(P*dxdt + Q*dydt, t))
