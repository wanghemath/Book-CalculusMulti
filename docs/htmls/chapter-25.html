import numpy as np

rng = np.random.default_rng(0)
X = rng.normal(size=(100, 2))
true_w = np.array([2.0, -1.0])
y = X @ true_w + 0.1*rng.normal(size=100)
w = np.zeros(2)
eta = 0.1
for k in range(100):
    grad = 2/len(y) * X.T @ (X @ w - y)
    w -= eta * grad
print(w)
