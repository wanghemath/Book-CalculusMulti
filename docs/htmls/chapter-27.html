import numpy as np

def finite_grad(f, x, h=1e-6):
    x = np.array(x, dtype=float)
    g = np.zeros_like(x)
    for j in range(len(x)):
        e = np.zeros_like(x); e[j] = 1
        g[j] = (f(x+h*e) - f(x-h*e))/(2*h)
    return g

f = lambda x: x[0]**2 + 3*x[1]**2
print(finite_grad(f, [1,2]))
